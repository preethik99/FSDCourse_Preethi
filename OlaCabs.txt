Create table Driver{
   DriverID int Primary Key,

}


Ride:                                                               1 to M Relationship

Create table Ride{
   RideID int Primary Key,
   UserID int Foreign Key,
   DriverID int Foreign Key,
   Fare int,
   RideStatus VarChar(100) 
 }
